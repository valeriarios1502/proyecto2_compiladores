fn maximo < type < T > > (a : T, b : T) : T {
  if (a > b) then {
    return a;
  } else then {
    return b;
  }
}

fn swap < type < T > > (a : *T, b : *T) : void {
  var tmp = *a;
  *a = *b;
  *b = tmp;
}

pub fn main () void {
  var x = maximo(3, 7);
  print(x);

  var y = maximo(10, 2);
  print(y);

  var z = maximo(100, 99);
  print(z);

  var a = 5;
  var b = 9;
  swap(&a, &b);
  print(a);
  print(b);

  var c = 42;
  var d = 0;
  swap(&c, &d);
  print(c);
  print(d);
}
